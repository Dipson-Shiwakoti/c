Odd or Even

CODE

            #include 

                int main() {
                    int num;
                    
                    printf("Enter a number: ");
                    scanf("%d", &num);
                    
                    if (num % 2 == 0)
                        printf("%d is even.\n", num);
                    else
                        printf("%d is odd.\n", num);
                
                    return 0;
                }

Using strcpy

CODE

            #include <stdio.h>
            #include <string.h>
                
                int main() {
                    char source[] = "Hello, World!";
                    char destination[50];
                    
                    strcpy(destination, source);
                    printf("Source: %s\n", source);
                    printf("Destination: %s\n", destination);
                    return 0;
                }

OUTPUT

Using strcat

CODE

            #include <stdio.h>
            #include <string.h>
                
                int main() {
                    char str1[50] = "Hello, ";
                    char str2[] = "World!";
                
                    strcat(str1, str2);
                    printf("Concatenated string: %s\n", str1);
                    
                    return 0;
                }

OUTPUT

Using strcmp

CODE

            #include <stdio.h>
            #include <string.h>
                
                int main() {
                    char str1[] = "Hello";
                    char str2[] = "Hello";
                
                    if (strcmp(str1, str2) == 0)
                        printf("Strings are equal\n");
                    else
                        printf("Strings are not equal\n");
                
                    return 0;
                }
